# 코드

```c
#include <msp430.h> 

/**
 * main.c
 */

int main(void){
    WDTCTL = WDTPW | WDTHOLD;   // stop watchdog timer

    P1OUT |= BIT1;
    P1REN |= BIT1;

    P2OUT |= BIT1;
    P2REN |= BIT1;

    P2DIR |= (BIT5 | BIT4);
    P2SEL |= (BIT5 | BIT4);

    TA2CTL = TASSEL_2 + MC_1;
    TA2CCR0 = 1000;
    TA2CCTL2 = OUTMOD_6;
    TA2CCR2 = 0;
    TA2CCTL1 = OUTMOD_6;
    TA2CCR1 = 0;

    __bis_SR_register(GIE);

    while(1)
    {
        if((P1IN & BIT1) == 0 && (P2IN & BIT1) != 0){  // P1.1 
                if(TA2CCR1 < 1000){
                    TA2CCR1 = TA2CCR1 + 1;
                }
                else{
                    TA2CCR1 = 1000;
                }
                __delay_cycles(10000);
            }

        else if((P1IN & BIT1) != 0 && (P2IN & BIT1) == 0){  // P2.1 
            if(TA2CCR2 < 1000){
                TA2CCR2 = TA2CCR2 + 1;
            }
            else{
                TA2CCR2 = 1000;
            }
            __delay_cycles(10000);
        }

        else if((P1IN & BIT1) == 0 && (P2IN & BIT1) == 0){ // both switch
            TA2CCR1 = 0;
            TA2CCR2 = 0;
            __delay_cycles(10000);
        }
    }
    return 0;
}

```
